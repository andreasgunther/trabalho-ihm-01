
-- criação de tabelas
CREATE TABLE departamentos(
  codigo INT AUTO_INCREMENT,
  descricao VARCHAR(200),
  CONSTRAINT pk_departamentos PRIMARY KEY (codigo)
);


DESCRIBE departamentos;

CREATE TABLE funcionarios(
  codigo INT AUTO_INCREMENT,
  nome VARCHAR(100),
  rua VARCHAR(100),
  numero INT,
  telefone VARCHAR(20),
  cod_departamento INT,
  CONSTRAINT pk_funcionarios PRIMARY KEY (codigo),
  CONSTRAINT fk_departamentos FOREIGN KEY (cod_departamento) REFERENCES departamentos (codigo)
);

DESCRIBE funcionarios;

CREATE TABLE dependentes(
  codigo INT AUTO_INCREMENT,
  nome VARCHAR(100),
  data_nascimento DATE,
  cod_funcionarios INT,
  CONSTRAINT pk_dependentes PRIMARY KEY (codigo),
  CONSTRAINT fk_funcionarios FOREIGN KEY (cod_funcionarios) REFERENCES funcionarios (codigo)
);

DESCRIBE dependentes;
